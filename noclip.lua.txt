--// NoClip Script with UI + Toggle Menu (KRNL)
--// Code by ChatGPT, Custom by Thịnh

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local StarterGui = game:GetService("StarterGui")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

local noclip = false
local menuVisible = true -- Trạng thái menu

-- Gửi thông báo khi bật script
StarterGui:SetCore("SendNotification", {
    Title = "Thông Báo";
    Text = "By: Thịnh";
    Duration = 5;
})

--// UI
local ScreenGui = Instance.new("ScreenGui")
local Frame = Instance.new("Frame")
local Button = Instance.new("TextButton")
local Label = Instance.new("TextLabel")

ScreenGui.Parent = game.CoreGui
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Frame (Menu)
Frame.Parent = ScreenGui
Frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
Frame.Position = UDim2.new(0.1, 0, 0.1, 0)
Frame.Size = UDim2.new(0, 220, 0, 130)
Frame.Active = true
Frame.Draggable = true
Frame.BorderSizePixel = 0
Frame.BackgroundTransparency = 0.1

-- Label
Label.Parent = Frame
Label.BackgroundTransparency = 1
Label.Size = UDim2.new(1, 0, 0.2, 0)
Label.Position = UDim2.new(0, 0, 0, 0)
Label.Font = Enum.Font.SourceSansBold
Label.Text = "NoClip Menu"
Label.TextColor3 = Color3.fromRGB(255, 255, 0)
Label.TextScaled = true

-- Nút bật/tắt noclip
Button.Parent = Frame
Button.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
Button.Position = UDim2.new(0.1, 0, 0.4, 0)
Button.Size = UDim2.new(0.8, 0, 0.4, 0)
Button.Font = Enum.Font.SourceSansBold
Button.TextColor3 = Color3.fromRGB(255, 255, 255)
Button.TextScaled = true
Button.Text = "NoClip: OFF"
Button.BorderSizePixel = 0

-- Toggle NoClip
local function toggleNoClip()
    noclip = not noclip
    if noclip then
        Button.BackgroundColor3 = Color3.fromRGB(150, 50, 50)
        Button.Text = "NoClip: ON"
    else
        Button.BackgroundColor3 = Color3.fromRGB(50, 150, 50)
        Button.Text = "NoClip: OFF"
    end
end

Button.MouseButton1Click:Connect(toggleNoClip)

-- Toggle Menu (Ẩn/Hiện UI bằng phím M)
UserInputService.InputBegan:Connect(function(input, gp)
    if gp then return end
    if input.KeyCode == Enum.KeyCode.M then
        menuVisible = not menuVisible
        Frame.Visible = menuVisible
    end
end)

-- Noclip function
RunService.Stepped:Connect(function()
    if noclip and character then
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = false
            end
        end
    end
end)
