-- Utility safe pcall HttpGet
local function http_get(url)
    local ok, res = pcall(function() return game:HttpGet(url) end)

-- Try load Orion UI
local OrionLib
do
    local orion_url = "https://raw.githubusercontent.com/shlexware/Orion/main/source" -- common hosting
    local ok, src = pcall(function() return game:HttpGet(orion_url) end)
    if ok and type(src)=="string" and #src>50 then
        local f,err = loadstring(src)
        if f then
            local success,lib = pcall(f)
            if success and lib then
                OrionLib = lib
            end
        end
    end
end

-- Fallback minimal UI creator (if Orion not loaded)
local function makeFallbackUI()
    local ScreenGui = Instance.new("ScreenGui")
    ScreenGui.Name = "FallbackUI"
    ScreenGui.ResetOnSpawn = false
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")

    local function makeFrame(props)
        local f = Instance.new("Frame")
        for k,v in pairs(props or {}) do f[k]=v end
        Instance.new("UICorner", f).CornerRadius = UDim.new(0,8)
        return f
    end
    local function makeBtn(parent, text, pos)
        local b = Instance.new("TextButton")
        b.Size = UDim2.new(1, -20, 0, 36)
        b.Position = pos or UDim2.new(0,10,0,10)
        b.BackgroundColor3 = THEME_COLOR
        b.TextColor3 = Color3.fromRGB(255,255,255)
        b.Text = text
        b.Font = Enum.Font.SourceSansBold
        b.TextSize = 16
        b.Parent = parent
        Instance.new("UICorner", b).CornerRadius = UDim.new(0,6)
        return b
    end

    -- Minimal layout: left menu + right content
    local main = makeFrame{Size=UDim2.new(0,520,0,320), Position=UDim2.new(0.5,-260,0.5,-160), BackgroundColor3 = Color3.fromRGB(35,20,60)}
    main.Parent = ScreenGui

    local menu = makeFrame{Size=UDim2.new(0,160,1,0), Position=UDim2.new(0,0,0,0), BackgroundColor3 = Color3.fromRGB(50,30,80)}
    menu.Parent = main

    local content = makeFrame{Size=UDim2.new(1,-160,1,0), Position=UDim2.new(0,160,0,0), BackgroundColor3 = Color3.fromRGB(60,40,90)}
    content.Parent = main

    return {
        ScreenGui = ScreenGui,
        Main = main,
        Menu = menu,
        Content = content,
        MakeButton = makeBtn
    }
end

    label.AnchorPoint = Vector2.new(0.5,0.5)

    label.TextColor3 = Color3.fromRGB(255,255,255)

    label.TextScaled = true
    label.Text = text

            label.TextTransparency = i/15
            label.BackgroundTransparency = 0.5 + i*(0.5/15)
            wait(0.03)
        end
        sg:Destroy()
    end)

-- THÔNG THƯỜNG TAB (noclip, fling, invisible, swim, wallstick)
-- --------------------
local toggles = {
    noclip = false,
    invisible = false,
    fling = false,
    swim = false,
    wallstick = false
}
local connections = {}

-- Noclip: set CanCollide false for character parts every frame
local function setNoclip(state)
    toggles.noclip = state
    if state then
        connections.noclip = RunService.Stepped:Connect(function()
            local char = LocalPlayer.Character
            if char then
                for _,part in ipairs(char:GetDescendants()) do
                    if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                        part.CanCollide = false
                    end
                end
            end
        end)
    else
        if connections.noclip then connections.noclip:Disconnect(); connections.noclip=nil end
        local char = LocalPlayer.Character
        if char then
            for _,part in ipairs(char:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = true
                end
            end
        end
    end
end

-- Invisible: set local character transparency
local function setInvisible(state)
    toggles.invisible = state
    local char = LocalPlayer.Character
    if not char then return end
    for _,p in ipairs(char:GetDescendants()) do
        if p:IsA("BasePart") and p.Name ~= "HumanoidRootPart" then
            p.LocalTransparencyModifier = state and 1 or 0
        elseif p:IsA("Decal") then
            p.Transparency = state and 1 or 0
        end
    end
end

-- Fling: apply a big velocity impulse while touching a player (simple approach)
local flingConnection
local function setFling(state)
    toggles.fling = state
    if state then
        local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if root then
            flingConnection = root.Touched:Connect(function(hit)
                local other = hit and hit.Parent
                if other and other ~= LocalPlayer.Character and other:FindFirstChild("Humanoid") then
                    pcall(function()
                        if root and root:IsA("BasePart") then
                            root.Velocity = Vector3.new(0,200,0)
                        end
                    end)
                end
            end)
        end
    else
        if flingConnection then flingConnection:Disconnect(); flingConnection=nil end
    end
end

-- Swim (bơi đi) -> simulate upward swim by applying BodyVelocity while holding toggle
local swimBV
local function setSwim(state)
    toggles.swim = state
    local char = LocalPlayer.Character
    if state then
        if char and char:FindFirstChild("HumanoidRootPart") then
            local root = char.HumanoidRootPart
            swimBV = Instance.new("BodyVelocity")
            swimBV.MaxForce = Vector3.new(0,400000,0)
            swimBV.Velocity = Vector3.new(0,50,0)
            swimBV.Parent = root
        end
    else
        if swimBV then swimBV:Destroy(); swimBV=nil end
    end
end

-- Wallstick: set HumanoidRootPart.Anchored while near wall (simple sticky)
local wallConn
local function setWallstick(state)
    toggles.wallstick = state
    if state then
        wallConn = RunService.Heartbeat:Connect(function()
            local char = LocalPlayer.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local root = char.HumanoidRootPart
                local ray = Ray.new(root.Position, root.CFrame.LookVector * 3)
                local hit = workspace:FindPartOnRayWithIgnoreList(ray, {char})
                if hit then
                    root.Anchored = true
                else
                    root.Anchored = false
                end
            end
        end)
    else
        if wallConn then wallConn:Disconnect(); wallConn=nil end
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.Anchored = false
        end
    end
end

-- --------------------
-- ESP TAB (name + distance)
-- --------------------
local espState = {player = false, name = false, distance = false}
local espGuis = {} -- map player -> billboard

local function createESPForPlayer(plr)
    if plr == LocalPlayer or espGuis[plr] then return end
    local char = plr.Character
    if not char then return end
    local root = char:FindFirstChild("Head") or char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    local bg = Instance.new("BillboardGui")
    bg.Name = "SimpleESP"
    bg.Adornee = root
    bg.Size = UDim2.new(0,200,0,40)
    bg.StudsOffset = Vector3.new(0,2.4,0)
    bg.AlwaysOnTop = true
    bg.Parent = CoreGui -- Use CoreGui to survive some UI layers; if blocked, will set Parent to RootGui
    -- Label inside
    local label = Instance.new("TextLabel", bg)
    label.BackgroundTransparency = 1
    label.Size = UDim2.new(1,0,1,0)
    label.Font = Enum.Font.GothamBold
    label.TextSize = 14
    label.TextColor3 = Color3.fromRGB(255,255,255)
    label.TextStrokeTransparency = 0.6

    espGuis[plr] = {gui = bg, label = label}
end

local function removeESPForPlayer(plr)
    if espGuis[plr] then
        pcall(function() espGuis[plr].gui:Destroy() end)
        espGuis[plr] = nil
    end
end

local function updateAllESP()
    for plr,data in pairs(espGuis) do
        local ok, _ = pcall(function()
            if not plr.Character or not plr.Character.PrimaryPart then
                removeESPForPlayer(plr); return
            end
            local label = data.label
            local textParts = {}
            if espState.name then table.insert(textParts, plr.Name) end
            if espState.distance and plr.Character and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and plr.Character:FindFirstChild("HumanoidRootPart") then
                local dist = (LocalPlayer.Character.HumanoidRootPart.Position - plr.Character.HumanoidRootPart.Position).Magnitude
                table.insert(textParts, string.format("%.0f m", dist))
            end
            label.Text = table.concat(textParts, " | ")
        end)
    end
end

-- Update esp when players join/leave/character added
Players.PlayerAdded:Connect(function(p)
    if espState.player then createESPForPlayer(p) end
end)
Players.PlayerRemoving:Connect(function(p)
    removeESPForPlayer(p)
end)

-- periodic update
spawn(function()
    while true do
        if espState.player or espState.name or espState.distance then
            -- ensure all players have guis when player esp enabled
            for _,plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and espState.player then
                    if not espGuis[plr] then createESPForPlayer(plr) end
                elseif not espState.player then
                    -- if player esp off, but name/distance on, still show (we treat player toggle as overall)
                    if not espState.name and not espState.distance then removeESPForPlayer(plr) end
                end
            end
            updateAllESP()
        else
            -- destroy all esp guis
            for p,_ in pairs(espGuis) do removeESPForPlayer(p) end
        end
        wait(0.25)
    end
end)
