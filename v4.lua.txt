-- Simple self-contained UI functions (minimal, avoids external libs)
local function createScreenGui(name)
    local sg = Instance.new("ScreenGui")
    sg.Name = name or "ScreenGui"
    sg.ResetOnSpawn = false
    sg.Parent = LocalPlayer:WaitForChild("PlayerGui")
    return sg
end

local function makeFrame(parent, props)
    local f = Instance.new("Frame")
    for k,v in pairs(props or {}) do f[k]=v end
    Instance.new("UICorner", f).CornerRadius = UDim.new(0,8)
    f.Parent = parent
    return f
end

local function makeButton(parent, text, props)
    local b = Instance.new("TextButton")
    b.Text = text or "Button"
    b.Font = Enum.Font.GothamBold
    b.TextSize = 16
    for k,v in pairs(props or {}) do b[k]=v end
    b.Parent = parent
    Instance.new("UICorner", b).CornerRadius = UDim.new(0,6)
    return b
end

-- Common tab functions (noclip, invisible, fling, swim, wallstick)
local toggles = {noclip=false, invisible=false, fling=false, swim=false, wallstick=false}
local connections = {}
local function setNoclip(state)
    toggles.noclip = state
    if state then
        connections.noclip = RunService.Stepped:Connect(function()
            local char = LocalPlayer.Character
            if char then for _,p in ipairs(char:GetDescendants()) do if p:IsA("BasePart") and p.Name~="HumanoidRootPart" then p.CanCollide=false end end end
        end)
    else
        if connections.noclip then connections.noclip:Disconnect(); connections.noclip=nil end
        local char = LocalPlayer.Character
        if char then for _,p in ipairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide=true end end end
    end
end

local function setInvisible(state)
    toggles.invisible = state
    local char = LocalPlayer.Character; if not char then return end
    for _,p in ipairs(char:GetDescendants()) do
        if p:IsA("BasePart") and p.Name~="HumanoidRootPart" then p.LocalTransparencyModifier = state and 1 or 0
        elseif p:IsA("Decal") then p.Transparency = state and 1 or 0 end
    end
end

local flingConn
local function setFling(state)
    toggles.fling = state
    if state then
        local root = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if root then
            flingConn = root.Touched:Connect(function(hit)
                local other = hit and hit.Parent
                if other and other~=LocalPlayer.Character and other:FindFirstChild("Humanoid") then pcall(function() if root and root:IsA("BasePart") then root.Velocity = Vector3.new(0,200,0) end end) end
            end)
        end
    else if flingConn then flingConn:Disconnect(); flingConn=nil end end
end

local swimBV
local function setSwim(state)
    toggles.swim = state
    local char = LocalPlayer.Character
    if state then
        if char and char:FindFirstChild("HumanoidRootPart") then
            local root = char.HumanoidRootPart
            swimBV = Instance.new("BodyVelocity"); swimBV.MaxForce = Vector3.new(0,400000,0); swimBV.Velocity = Vector3.new(0,50,0); swimBV.Parent = root
        end
    else if swimBV then swimBV:Destroy(); swimBV=nil end end
end

local wallConn
local function setWallstick(state)
    toggles.wallstick = state
    if state then
        wallConn = RunService.Heartbeat:Connect(function()
            local char = LocalPlayer.Character
            if char and char:FindFirstChild("HumanoidRootPart") then
                local root = char.HumanoidRootPart
                local ray = Ray.new(root.Position, root.CFrame.LookVector * 3)
                local hit = workspace:FindPartOnRayWithIgnoreList(ray, {char})
                if hit then root.Anchored = true else root.Anchored = false end
            end
        end)
    else if wallConn then wallConn:Disconnect(); wallConn=nil end
        local char = LocalPlayer.Character if char and char:FindFirstChild("HumanoidRootPart") then char.HumanoidRootPart.Anchored = false end
    end
end

-- ESP system
local espState = {player=false, name=false, distance=false}
local espGuis = {}

local function createESPForPlayer(plr)
    if plr==LocalPlayer or espGuis[plr] then return end
    local char = plr.Character if not char then return end
    local root = char:FindFirstChild("Head") or char:FindFirstChild("HumanoidRootPart") if not root then return end
    local bg = Instance.new("BillboardGui"); bg.Name="SimpleESP"; bg.Adornee=root; bg.Size=UDim2.new(0,200,0,40); bg.StudsOffset=Vector3.new(0,2.4,0); bg.AlwaysOnTop=true
    local ok,err = pcall(function() bg.Parent = CoreGui end) if not ok then bg.Parent = LocalPlayer:WaitForChild("PlayerGui") end
    local label = Instance.new("TextLabel", bg); label.BackgroundTransparency=1; label.Size=UDim2.new(1,0,1,0); label.Font=Enum.Font.GothamBold; label.TextSize=14; label.TextColor3=Color3.fromRGB(255,255,255); label.TextStrokeTransparency=0.6
    espGuis[plr] = {gui=bg, label=label}
end

local function removeESPForPlayer(plr) if espGuis[plr] then pcall(function() espGuis[plr].gui:Destroy() end); espGuis[plr]=nil end end

local function updateAllESP()
    for plr,data in pairs(espGuis) do pcall(function()
        if not plr.Character or not plr.Character.PrimaryPart then removeESPForPlayer(plr); return end
        local label = data.label local textParts = {}
        if espState.name then table.insert(textParts, plr.Name) end
        if espState.distance and plr.Character and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") and plr.Character:FindFirstChild("HumanoidRootPart") then
            local dist = (LocalPlayer.Character.HumanoidRootPart.Position - plr.Character.HumanoidRootPart.Position).Magnitude
            table.insert(textParts, string.format("%.0f m", dist))
        end
        label.Text = table.concat(textParts, " | ")
    end) end
end

Players.PlayerAdded:Connect(function(p) if espState.player then createESPForPlayer(p) end end)
Players.PlayerRemoving:Connect(function(p) removeESPForPlayer(p) end)

spawn(function()
    while true do
        if espState.player or espState.name or espState.distance then
            for _,plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and espState.player then if not espGuis[plr] then createESPForPlayer(plr) end
                elseif not espState.player then if not espState.name and not espState.distance then removeESPForPlayer(plr) end end
            end
            updateAllESP()
        else for p,_ in pairs(espGuis) do removeESPForPlayer(p) end end
        wait(0.25)
    end
end)

-- SERVER HOP
local function fetchServers(placeId,cursor)
    local url = "https://games.roblox.com/v1/games/"..tostring(placeId).."/servers/Public?sortOrder=Asc&limit=100"
    if cursor and cursor~="" then url = url .. "&cursor=" .. tostring(cursor) end
    local ok,res = pcall(function() return HttpService:GetAsync(url) end)
    if not ok or not res then return nil end
    local suc,data = pcall(function() return HttpService:JSONDecode(res) end)
    if not suc or not data then return nil end
    return data
end

local function serverHopRandom()
    spawn(function()
        local placeId = game.PlaceId
        local data = fetchServers(placeId)
        if not data then warn("Khong lay duoc danh sach server."); return end
        local servers = data.data or data if not servers or #servers==0 then warn("Khong co server."); return end
        local choices = {}
        for _,s in ipairs(servers) do if s.id then table.insert(choices,s) end end
        local pick = choices[math.random(1,#choices)]
        if pick and pick.id then pcall(function() TeleportService:TeleportToPlaceInstance(placeId, tostring(pick.id), LocalPlayer) end) end
    end)
end

local function serverHopLowPlayer()
    spawn(function()
        local placeId = game.PlaceId
        local data = fetchServers(placeId)
        if not data then warn("Khong lay duoc danh sach server."); return end
        local servers = data.data or data
        table.sort(servers, function(a,b) return (a.playing or 0) < (b.playing or 0) end)
        local pick = nil for _,s in ipairs(servers) do if s.id then pick = s break end end
        if pick and pick.id then pcall(function() TeleportService:TeleportToPlaceInstance(placeId, tostring(pick.id), LocalPlayer) end) end
    end)
end
